const SS = SpreadsheetApp.getActiveSpreadsheet(); function doGet(e) { const action = e.parameter.action; if (action === 'read') return leer(e); return ContentService.createTextOutput(JSON.stringify({ok:false,msg:'Acción no válida'})) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.action === 'write') return escribir(data); if (data.action === 'readAll') return leerTodo(data); return resp({ok:false,msg:'Acción desconocida'}); } catch(err) { return resp({ok:false,msg:err.toString()}); } } function escribir(data) { const hoja = obtenerHoja('Datos'); const fila = [ new Date().toISOString(), data.cargo, data.asesor, data.mes, data.evaluador, data.semana, data.indId, data.entregable, data.meta, data.logrado, data.cumpl, data.nota, data.obs ]; // Buscar si ya existe fila con mismo key const rows = hoja.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][1]===data.cargo && rows[i][2]===data.asesor && rows[i][3]===data.mes && rows[i][4]===data.evaluador && rows[i][5]===data.semana && rows[i][6]===data.indId) { hoja.getRange(i+1, 1, 1, fila.length).setValues([fila]); return resp({ok:true,msg:'Actualizado'}); } } hoja.appendRow(fila); return resp({ok:true,msg:'Guardado'}); } function leer(e) { const hoja = obtenerHoja('Datos'); const rows = hoja.getDataRange().getValues(); const cargo=e.parameter.cargo, asesor=e.parameter.asesor, mes=e.parameter.mes, evaluador=e.parameter.evaluador, semana=e.parameter.semana; const result = rows.slice(1).filter(r => r[1]===cargo && r[2]===asesor && r[3]===mes && r[4]===evaluador && r[5]===semana ).map(r=>({indId:r[6],meta:r[8],logrado:r[9],cumpl:r[10],nota:r[11],obs:r[12]})); return ContentService.createTextOutput(JSON.stringify({ok:true,data:result})) .setMimeType(ContentService.MimeType.JSON); } function leerTodo(data) { const hoja = obtenerHoja('Datos'); const rows = hoja.getDataRange().getValues(); const result = rows.slice(1).filter(r => r[1]===data.cargo && r[2]===data.asesor && r[3]===data.mes ).map(r=>({evaluador:r[4],semana:r[5],indId:r[6],meta:r[8],logrado:r[9],cumpl:r[10],nota:r[11],obs:r[12]})); return resp({ok:true,data:result}); } function obtenerHoja(nombre) { let h = SS.getSheetByName(nombre); if (!h) { h = SS.insertSheet(nombre); h.appendRow(['Timestamp','Cargo','Asesor','Mes','Evaluador','Semana', 'IndID','Entregable','Meta','Logrado','Cumpl%','Nota','Observaciones']); h.getRange(1,1,1,13).setFontWeight('bold').setBackground('#1A3C6E').setFontColor('#FFFFFF'); } return h; } function resp(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
